```
def majorityElementII(arr):
    n = len(arr)
    d = {}
    for i in range(n):
        if arr[i] in d:
            d[arr[i]] += 1
        else:
            d[arr[i]] = 1

    l = []
    count = n // 3
    for i in d.keys():
        if d[i] > count:
            l.append(i)

    return l    
```
