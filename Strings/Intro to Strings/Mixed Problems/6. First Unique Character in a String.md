def findNonRepeating(s):
    visited = [0 for _ in range(26)]
    n =len(s)
    for i in range(n):
        number = ord(s[i]) - 97
        visited[number] += 1

    ch = "#"
    for i in range(n):
        if visited[ord(s[i])-97] == 1:
            return s[i]
    return ch
