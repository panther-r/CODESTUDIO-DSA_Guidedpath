```
def subArrayCount(nums, k):
    n = len(nums)
    res = {0: 1}
    prefSum = 0
    count = 0
    for i in range(n):
        prefSum += nums[i]
        rem = (prefSum) % k
        if rem < 0:
            rem += k
        if rem in res:
            count += res[rem]
        if rem not in res:
            res[rem] = 0
        res[rem] += 1
    return count
```
