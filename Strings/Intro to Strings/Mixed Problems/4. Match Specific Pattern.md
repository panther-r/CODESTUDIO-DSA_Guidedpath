def map(s,pattern,lp):   
    v1 = [0 for _ in range(256)] # for s
    v2 = [0 for _ in range(256)] # for pattern
    for i in range(lp):
        n1 = ord(s[i])  
        n2 = ord(pattern[i])
        if v1[n1] == 0 and v2[n2] == 0:
            v1[n1] = n2
            v2[n2] = n1
        elif (v2[n2] != n1 or v1[n1] != n2):
            return False
    return True
    
    

def matchSpecificPattern(words, n, pattern):
    ans = []
    lp = len(pattern)
    for i in range(n):
        if len(words[i]) == lp:
            if map(words[i],pattern,lp):
                ans.append(words[i])
    return ans
